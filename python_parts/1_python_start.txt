#nous avons encapsuler l'age
def _getage(self):
	try:
	   return self._age ou print("aucune recupération")
	except AttributeError:
	   print(l'age n'existe pas vous l'avez effacé")

def _setage(self, nouvelage):
	if nouvelage < 0:
		self._age=0
	else:
		self._age = nouvelage

def _delage(self):
	del self._age
